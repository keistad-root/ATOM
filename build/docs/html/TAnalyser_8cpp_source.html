696int calNIncludePixel(const TMatrix2D<int>* matrix) {
697 int centreX = 0, centreY = 0;
698 int clusterSize = 0;
699 for ( int pixelX = 0; pixelX < matrix->getNRow(); pixelX++ ) {
700 for ( int pixelY = 0; pixelY < matrix->getNColumn(); pixelY++ ) {
701 if ( matrix->getElement(pixelX, pixelY) == 1 ) {
702 centreX += pixelX * 2;
703 centreY += pixelY * 2;
704 clusterSize++;
705 }
706 }
707 }
708 int radiusSquare = 0;
709 for ( int pixelX = 0; pixelX < matrix->getNRow(); pixelX++ ) {
710 for ( int pixelY = 0; pixelY < matrix->getNColumn(); pixelY++ ) {
711 if ( matrix->getElement(pixelX, pixelY) == 1 ) {
712 int distance = pow(abs(2 * pixelX * clusterSize - centreX) + clusterSize, 2) + pow(abs(2 * pixelY * clusterSize - centreY) + clusterSize, 2);
713 radiusSquare = std::max(radiusSquare, distance);
714 }
715 }
716 }
717 int count = 0;
718 for ( int x = floor((centreX - sqrt(radiusSquare)) / (2 * clusterSize)); x < ceil((centreX + sqrt(radiusSquare)) / (2 * clusterSize)) + 1; x++ ) {
719 for ( int y = floor((centreY - sqrt(radiusSquare)) / (2 * clusterSize)); y < ceil((centreY + sqrt(radiusSquare)) / (2 * clusterSize)) + 1; y++ ) {
720 if ( static_cast<int>(std::pow(abs(2 * x * clusterSize - centreX) + clusterSize, 2)) + static_cast<int>(std::pow(abs(2 * y * clusterSize - centreY) + clusterSize, 2)) <= radiusSquare ) {
721 count++;
722 }
723 }
724 }
725 return count;
726}

728double calRatioOfRadius(const TMatrix2D<int>* matrix) {
729 int centreX = 0, centreY = 0;
730 int clusterSize = 0;
731 for ( int pixelX = 0; pixelX < matrix->getNRow(); pixelX++ ) {
732 for ( int pixelY = 0; pixelY < matrix->getNColumn(); pixelY++ ) {
733 if ( matrix->getElement(pixelX, pixelY) == 1 ) {
734 centreX += pixelX * 2;
735 centreY += pixelY * 2;
736 clusterSize++;
737 }
738 }
739 }
740 int longRadiusSquare = 0;
741 int shortRadiusSquare = 100000000;
742 for ( int pixelX = 0; pixelX < matrix->getNRow(); pixelX++ ) {
743 for ( int pixelY = 0; pixelY < matrix->getNColumn(); pixelY++ ) {
744 bool hasValue = matrix->getElement(pixelX, pixelY) == 1;
745 bool isWorldBorder = pixelX == 0 || pixelY == 0 || pixelX == matrix->getNRow() - 1 || pixelY == matrix->getNColumn() - 1;
746 bool isBorder = isWorldBorder ? true : matrix->getElement(pixelX - 1, pixelY) == 0 || matrix->getElement(pixelX, pixelY - 1) == 0 || matrix->getElement(pixelX + 1, pixelY) == 0 || matrix->getElement(pixelX, pixelY + 1) == 0;
747 if ( hasValue ) {
748 int distance = pow(2 * pixelX * clusterSize - centreX, 2) + pow(2 * pixelY * clusterSize - centreY, 2);
749 longRadiusSquare = std::max(longRadiusSquare, distance);
750 }
751 if ( (hasValue && isWorldBorder) || (hasValue && !isWorldBorder && isBorder) ) {
752 int distance = pow(2 * pixelX * clusterSize - centreX, 2) + pow(2 * pixelY * clusterSize - centreY, 2);
753 shortRadiusSquare = std::min(shortRadiusSquare, distance);
754 }
755 }
756 }
757 if ( matrix->getNRow() == 1 || matrix->getNColumn() == 1 ) {
758 shortRadiusSquare = 0;
759 }
760 return sqrt(static_cast<double>(shortRadiusSquare) / longRadiusSquare);
761}
