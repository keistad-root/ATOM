64bool Unit::seperate(std::vector<std::string>& store, std::string_view unit) {
65 int iStart = 0;
66 int iNow = 0;
67
68 bool doubleSlash = false;
69 bool isNum = false;
70
71 bool error = false;
72 for ( const char letter : unit ) {
73 if ( letter == '*' ) {
74 store.push_back(static_cast<std::string>(unit.substr(iStart, iNow - iStart)));
75 store.push_back(static_cast<std::string>(unit.substr(iNow, 1)));
76 iStart = iNow + 1;
77 } else if ( letter == '/' ) {
78 if ( doubleSlash ) {
79 std::cerr << "Unit notation to write '/' twice is forbidden" << std::endl;
80 error = true;
81 }
82 store.push_back(static_cast<std::string>(unit.substr(iStart, iNow - iStart)));
83 store.push_back(static_cast<std::string>(unit.substr(iNow, 1)));
84 iStart = iNow + 1;
85 doubleSlash = true;
86 } else if ( (letter == '-' || isdigit(letter)) && !isNum ) {
87 store.push_back(static_cast<std::string>(unit.substr(iStart, iNow - iStart)));
88 isNum = true;
89 iStart = iNow;
90 } else if ( isNum && !isdigit(letter) ) {
91 iStart = iNow;
92 isNum = false;
93 }
94 if ( iNow == unit.size() - 1 ) {
95 store.push_back(static_cast<std::string>(unit.substr(iStart)));
96 }
97 iNow++;
98 }
